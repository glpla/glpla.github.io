具体过程
good.json数据为例
在 静态资源服务器 的基础上，增加监听路由
let fs = require('fs');
let express = require('express')

// 创建Web服务器对象
let app = express();

// 静态资源处理
app.use(express.static('./public'));

// 读取JSON数据
let jsonStr = fs.readFileSync('./public/data/good.json', {
  encoding: 'utf8'
});

let data = JSON.parse(jsonStr);

//全部拉取
app.get('good/all', (req, res) => {
    res.send(data)
})

//请求格式 url/data?page=1&pageSize=10
app.get('/good/page', (req, res) => {
  let [ ...arr ] = data
  let start = (req.query.page - 1) * req.query.pageSize;
  res.setHeader('X-Total-Count', data.length);
  res.send(arr.splice(start, req.query.pageSize));
});

app.listen(3000, () => {
  console.log('服务器启动成功，地址为：http://127.0.0.1:3000');
});
