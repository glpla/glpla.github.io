String

Q1：什么是字符串？

. C中没有字符串类型

. 以'\0'为结束标记的连续字符的集合，以""括起来

. 结束标记按照char打印不出来；按照int打印为0

. 输出串时，应该判断是否是结束标记，而不是判断长度

Q1：如何表示一个串？

      char *str = "hello, cnplaman.";//指针形式：容易输出；但长度需要使用第三方函数strlen获取
      char str[] = "hello, cnplaman.";//数组形式：容易获取空间大小(包括结束标记'\0')
      char str[100] = "hello, cnplaman.";//数组形式
      char str[100] = {'1', '2', '3'};//字符数组；不建议。如果没有指定大小，则没有结束标记；输出会异常，甚至直到找到一个其它数据的结束标记才会停止

Q2：如何输出一个指针串？

      //方案1：
      printf("%s\n",str);
      //方案2：遍历
      while (*str0)
      {
        printf("%c", *str0);
        str0++;

demo1：逆序输出给定的字符串

      #include <stdio.h>
      #include <string.h>
      int main(void)
      {
        char str[] = "welcome";
        int i = 0;
        // int len = sizeof(str) / sizeof(char);8
        int len = strlen(str);//7
        for (int i = len - 1; i >= 0; i--)
        {
          //putchar(str[i]);忽略结束标记
          printf("%c\n", str[i]);//输出结束标记
        }
        return 0;
      }

demo2：逆序输出输入的字符串

      #include <stdio.h>
      #include <string.h>
      #define N 20
      int main(void)
      {
        char str[N];
        int i = 0;
        printf(">");
        gets(str);
        // int len = sizeof(str) / sizeof(char);
        int len = strlen(str);
        for (int i = len - 1; i >= 0; i--)
        {
          putchar(str[i]);
        }
        return 0;
      }

demo3：串的链接

      int main(void)
      {
        char str[20] = "welcome";
        char *p = "pla";
        int len = 0;
        while (str[len])
        {
          len++;
        }
        while (*p)
        {
          printf("%c\n", *p);
          *(str + len) = *p;
          len++;
          p++;
        }
        printf("%s\n", str);
