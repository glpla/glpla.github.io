串的基本使用 返回顶部

Q1：什么是字符串？

. C中没有字符串类型

. 以'\0'为结束标记的连续字符的集合，以""括起来

. 结束标记按照char打印不出来；按照int打印为0

. 输出串时，应该判断是否是结束标记，而不是判断长度

Q1：如何表示一个串？

      char *str = "hello, cnplaman.";//指针形式：容易输出；但长度需要使用第三方函数strlen获取
      char str[] = "hello, cnplaman.";//数组形式：容易获取空间大小(包括结束标记'\0')
      char str[100] = "hello, cnplaman.";//数组形式
      char str[100] = {'1', '2', '3'};//字符数组；不建议。如果没有指定大小，则没有结束标记；输出会异常，甚至直到找到一个其它数据的结束标记才会停止

demo1：逆序输出给定的字符串

      #include <stdio.h>
      #include <string.h>
      int main(void)
      {
        char str[] = "welcome";
        int i = 0;
        // int len = sizeof(str) / sizeof(char);8
        int len = strlen(str);//7
        for (int i = len - 1; i >= 0; i--)
        {
          //putchar(str[i]);忽略结束标记
          printf("%c\n", str[i]);//输出结束标记
        }
        return 0;
      }

顺序串 返回顶部

        #include <stdio.h>
        #include <stdlib.h>
        
        #define maxSize 20
        #define TRUE 1
        #define FALSE 0
        typedef struct
        {
            char data[maxSize];
            int len;
        } String;
        int assign(String *s, char *str)
        {
            int i;
            for (i = 0; *(str + i) != '\0'; i++)
            {
                i++;
            }
            printf("len=%d\n", i);
            if (i > maxSize)
            {
                return FALSE;
            }
            i = 0;
            while (*(str + i) != '\0')
            {
                s->data[i] = *(str + i);
                i++;
            }
            s->len = i;
            return TRUE;
        }
        String *substr(String *s, int i, int len)
        {
            String *str = (String *)malloc(sizeof(String));
            str->len = 0;
            if (i < 0 || i > s->len || len < 0)
            {
                return str;
            }
            // 这个变量 j 可以省略
            int j = 0;
            for (int ind = i; ind < i + len; ind++)
            {
                str->data[j] = s->data[ind];
                j++;
            }
            //MUST
            str->data[j] = '\0';
            str->len = j;
            return str;
        }
        
        int main(void)
        {
            char *str = "hi, there. boy";
            int res;
            String *s = (String *)malloc(sizeof(String));
            res = assign(s, str);
            if (res)
            {
                puts(s->data);
                printf("%d\n", s->len);
            }
            String *newstr = substr(s, 2, 4);
            puts(newstr->data);
            printf("%d\n", newstr->len);
            free(s);
            return 0;
        }

链式串 返回顶部

        typedef struct Node
        {
            char data;
            struct Node *next;
        } Node;

        typedef struct
        {
            Node *head;
            int len;
        } String;

        void assign(String *s, char *str)
        {
            Node *p, *pre = s->head;
            int i;
            for (i = 0; *(str + i) != '\0'; i++)
            {
                p = (Node *)malloc(sizeof(Node));
                p->data = *(str + i);
                p->next = pre->next;
                pre->next = p;
                // MUST
                pre = p;
            }
            s->len = i;
        }

        void display(String *s)
        {
            Node *p = s->head->next;
            while (p)
            {
                printf("%c", p->data);
                p = p->next;
            }
            printf("\n");
        }  

        int main(void)
        {
            String *s = (String *)malloc(sizeof(String));
            s->head = (Node *)malloc(sizeof(Node));
            s->head->data = '#';
            s->head->next = NULL;
            s->len = 0;
            assign(s, "hi, there.");
            printf("%d\n", s->len);
            display(s);
            free(s);
            return 0;
        }
