串的基本使用 

Q: 基本概念

串：0个或多个字符组成的有限序列；以双引号包裹，如”hi, there.”

串长：字符的个数（标记符号’\0’呢？）

空串：长度为0

串相等：当且仅当两个串的长度相等，且对应的字符也相等

子串：串中任意个连续的字符组成的序列；空串是任何串的子串

注意：

. C中没有字符串类型

. 输出串时，应该判断是否是结束标记，而不是判断长度

Q: 如何声明一个串？

      char *str = "hello, cnplaman.";//指针形式：容易输出；但长度需要使用第三方函数strlen获取
      char str[] = "hello, cnplaman.";//数组形式：容易获取空间大小(包括结束标记'\0')
      char str[100] = "hello, cnplaman.";//数组形式

Q: 如何输出一个指针串？

      //方案1: 格式化函数
      printf("%s\n",str);
      //方案2: 格式化函数
      puts(str);
      //方案3: 遍历
      while (*str0)
      {
        printf("%c", *str0);
        str0++;
      }

注意：

. 防止串丢失。。。

. 防止结束标记丢失

. 输出串时，应该判断是否是结束标记，而不是判断长度

串函数 

0. 头文件

<string.>

1. 串长

strlen(str)：返回指针指向串的长度

2. 比较串

strcmp(str1, str2)：先转换成ASCII码再比较，直到遇到不同的或者结束符；返回0，表示相等；返回负数，表示s1小于s2；返回正数，表示s1大于s2；类似的还有指定位数的比较strncmp(str0, str1, n)

3. 拷贝串

strcpy(des, src)

strncpy(des, src, n)：只拷贝n个字符过去，不包括结束标记

4. 拼接串

strcat(des, src)：des应有足够的长度来容纳拼接后的总长度；返回拼接后des的地址

strncat(str0, str1, n)，表示拼接时取后面串的多少位

顺序串 

        #include <stdio.h>
        #include <stdlib.h>

        #define maxSize 20
        #define TRUE 1
        #define FALSE 0
        #define ERROR -1
        typedef struct
        {
            char data[maxSize];
            int len;
        } String;

        int assign(String *s, char *str)
        {
            int i;
            for (i = 0; *(str + i) != '\0'; i++)
            {
                i++;
            }
            printf("len=%d\n", i);
            if (i > maxSize)
            {
                return ERROR;
            }
            i = 0;
            while (*(str + i) != '\0')
            {
                s->data[i] = *(str + i);
                i++;
            }
            // MUST
            s->data[i] = '\0';
            s->len = i;
            return TRUE;
        }

        void display(String *s)
        {
            puts(s->data);
        }

        String *substr(String *s, int i, int len)
        {
            String *str = (String *)malloc(sizeof(String));
            str->len = 0;
            if (i < 0 || i > s->len || len < 0)
            {
                return str;
            }
            // 这个变量可以省略
            int j = 0;
            for (int ind = i; ind < i + len; ind++)
            {
                str->data[j] = s->data[ind];
                j++;
            }
            // MUST
            str->data[j] = '\0';
            str->len = j;
            return str;
        }
        int indexOfCh(String *s, char ch)
        {
            if (!s->len)
            {
                return ERROR;
            }
            int ind;
            for (int i = 0; i < s->len; i++)
            {
                if (s->data[i] == ch)
                {
                    return i;
                }
            }
            return ERROR;
        }
        int match(String *s, String *p, int s_start, int p_start, int *s_fail, int *p_fail)
        {
            int i = s_start, j = p_start;
            // 从p串的第一位开始比较
            for (; j < p->len; i++, j++)
            {
                if (s->data[i] != p->data[j])
                {
                    *s_fail = i;
                    *p_fail = j;
                    return FALSE;
                }
            }
            return TRUE;
        }

        int indexOfStr(String *s, String *p, int pos)
        {
            int s_start = 0, p_start = 0, s_fail, p_fail;
            for (s_start = pos; s_start <= s->len - p->len; s_start++)
            {
                if (match(s, p, s_start, p_start, &s_fail, &p_fail))
                {
                    return s_start;
                }
            }
            return FALSE;
        }

        int main(void)
        {
            char *str0 = "hi, there. boy";
            char *str1 = "here";
            int res;
            String *s = (String *)malloc(sizeof(String));
            String *p = (String *)malloc(sizeof(String));

        }

链式串 

        typedef struct Node
        {
            char data;
            struct Node *next;
        } Node;

        typedef struct
        {
            Node *head;
            int len;
        } String;

        void assign(String *s, char *str)
        {
            Node *p, *pre = s->head;
            int i;
            for (i = 0; *(str + i) != '\0'; i++)
            {
                p = (Node *)malloc(sizeof(Node));
                p->data = *(str + i);
                p->next = pre->next;
                pre->next = p;
                // MUST
                pre = p;
            }
            s->data[i] = '\0';
            s->len = i;
        }

        void display(String *s)
        {
            Node *p = s->head->next;
            while (p)
            {
                printf("%c", p->data);
                p = p->next;
            }
            printf("\n");
        }  

        int main(void)
        {
            String *s = (String *)malloc(sizeof(String));
            s->head = (Node *)malloc(sizeof(Node));
            s->head->data = '#';
            s->head->next = NULL;
            s->len = 0;
            assign(s, "hi, there.");
            printf("%d\n", s->len);
            display(s);
            free(s);
            return 0;
        }
