List

      #include <stdio.h>
      #include <stdlib.h>
      #define maxSize 20
      
      typedef struct
      {
        int *data;
        int len;
        int size;
      } List;

      void init(List *list)
      {
        list->data = (int *)malloc(maxSize * sizeof(int));
        if (!list->data)
        {
          printf("init fail\n");
          exit(0);
        }
        list->len = 0;
        list->size = maxSize;
        printf("list length=%d\n", list->len);
      }

      void toString(List *list)
      {
        printf("list start\n");
        for (int i = 0; i < list->len; i++)
        {
          printf("%d\n", list->data[i]);
        }
        printf("list end\n");
      }
      
      void sizeOfList(List *list)
      {
        printf("size=%d\n", list->len);
      }

      void insert(List *list, int el)
      {
        printf("before size=%d\n", list->len);
        if (list->len == maxSize)
        {
          return;
        }
        list->data[list->len] = el;
        list->len++;
        printf("after size=%d\n", list->len);
      }
      
      int main(void)
      {
        List list;
        init(&list);
        toString(&list);
        sizeOfList(&list);
        insert(&list, 1);
        insert(&list, 2);
        insert(&list, 3);
        insert(&list, 4);
        insert(&list, 5);
        toString(&list);
        return 0;
      }
