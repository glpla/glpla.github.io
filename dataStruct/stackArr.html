Stack With Array

    class StackArr {
      constructor() {
        this.items = [];
      }
      push(el) {
        this.items.push(el)
      }
      pop() {
        this.items.pop()
      }
      top() {
        return this.items[this.items.length - 1]
      }
      isEmpty() {
        return this.items.length == 0
      }
      size() {
        return this.items.length
      }
      toString() {
        return this.items.join('-')
      }
      clear() {
        this.items = []
      }
    }
