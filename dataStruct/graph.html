边节点数据结构

    typedef struct eNode
    {
        int adjV;
        int w;
        struct eNode *nextAdjV;
    } ENode;
图数据结构定义

    typedef struct
    {
        int vSize;
        int eSize;
        ENode **arr;
    } Graph;
图的初始化

    void init(Graph *g, int size)
    {
        int i;
        g->vSize = size;
        g->eSize = 0;
        g->arr = (ENode *)malloc(g->eSize * sizeof(ENode));
        if (g->arr)
        {
            for (i = 0; i < g->vSize; i++)
            {
                g->arr[i]->nextAdjV = NULL;
            }
        }
    }
图的销毁

    void destroy(Graph *g)
    {
        int i;
        ENode *p, *q;
        for (i = 0; i < g->vSize; i++)
        {
            p = g->arr[i];
            q = p;
            while (q)
            {
                p = p->nextAdjV;
                free(q);
                q = p;
            }
        }
        free(g->arr);
    }
