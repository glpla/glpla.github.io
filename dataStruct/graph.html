    {
        int vSize;
        int eSize;
        ENode **arr;
    } Graph;
图的初始化
根据顶点数创建一维数组，并把每个元素的边节点指向设为空；

    void init(Graph *g, int size)
    {
        int i;
        g->vSize = size;
        g->eSize = 0;
        g->arr = (ENode *)malloc(g->eSize * sizeof(ENode));
        if (g->arr)
        {
            for (i = 0; i < g->vSize; i++)
            {
                g->arr[i]->nextAdjV = NULL;
            }
        }
    }
图的销毁
销毁一维数组；
销毁所有边节点；

    void destroy(Graph *g)
    {
        int i;
        ENode *p, *q;
        for (i = 0; i < g->vSize; i++)
        {
            p = g->arr[i];
            q = p;
            while (q)
            {
                p = p->nextAdjV;
                free(q);
                q = p;
            }
        }
        free(g->arr);
    }
在某个顶点vertex中查找边edge
实际上查找的是和顶点邻接的顶点；找到顶点了，就找到边了；
请自行补充参数的有效性验证；

    int edgeFind(Graph *g, int ind, int vertex)
    {
        // the validation of ind and vertex
        // by yourself
    
        ENode *p = g->arr[ind];
        while (p && p->adjV != vertex)
        {
            p = p->nextAdjV;
        }
        if (!p)
        {
            return -1;
        }
        else
        {
            return 0;
        };
    }
在某个顶点vertex中插入边edge
请自行补充参数的有效性验证；

    int edgeInsert(Graph *g, int ind, int vertex, int w)
    {
        // the validation of n and vertex
        // by yourself
    
        if (!edgeFind(g, ind, vertex))
        {
            return -1;
        }
        ENode *p = (ENode *)malloc(sizeof(ENode));
        p->adjV = ind;
        p->w = w;
        p->nextAdjV = g->arr[ind];
        g->arr[ind] = p;
        g->eSize++;
        return 0;
    }

    int edgeDel(Graph *g, int ind, int vertex)
    {
        // the validation of n and vertex
        // by yourself
        ENode *p, *q;
        p = g->arr[ind];
        q = NULL;
        while (p && p->nextAdjV != vertex)
        {
            q = p;
            p = p->nextAdjV;
        }
        if (!p)
        {
            return -1;
        }
        else
        {
            q->nextAdjV = p->nextAdjV;
            free(p);
            g->eSize--;
            return 0;
        }
    }
