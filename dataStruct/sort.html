排序Sort

    #include <stdio.h>
    #include <stdlib.h>
    int main(void)
    {
    int a[] = {12, 8, 9, 78, 65, 4, 0, 21, 7}, n, i, j, t;
    n = sizeof(a) / sizeof(int);
    for (i = 0; i < n - 1; i++)
    {
        for (j = 0; j < n - 1 - i; j++)//如何找边界？
        {
        if (a[j] < a[j + 1])//如果是大于，则为升序
        {
            t = a[j];
            a[j] = a[j + 1];
            a[j + 1] = t;
        }
        }
    }
    for (i = 0; i < n; i++)
    {
        printf("%d\n", a[i]);
    }
    return 0;
    }
