import { useMenuStore } from '@/stores/menu'

    router.beforeEach((to, from, next) => {
        // 其它逻辑
    
        const store = useMenuStore();
        store.flag = false;
        next();
    })
    
    router.afterEach((to, from) => {
        // 其它逻辑
    
        if (to.path === from.path) {
            const store = useMenuStore()
            store.flag = false
        }
    })
