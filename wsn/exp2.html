UART

实操部分
实验目的
1. 掌握CC 2530 UART串口寄存器设置；
2. 掌握UART串口中断函数程序的编程方法；
实验内容
在IAR集成开发环境中编写定时器中断程序；通过按键以串行方式发送指定数据；
预备知识
1. 串口通信；
2. 了解C语言的基本知识；
3. 了解IAR中编写和调试程序的方法；
实验需求
安装IAR和相关驱动的个人PC1台
mini USB线1根
仿真器CC Debugger1根
ZigBee通用节点板1块
参考源码

#include "ioCC2530.h"
#include "../lib/delay.h"

char uart_buffer;

//串口从IO接收要发送的数据
void UartTX_Send_String( unsigned char *Data, int len )
{
    for( int j = 0; j < len; j++ )
    {
        //USART x收发数据缓冲区寄存器
        U0DBUF = *Data++;
        //如果发送完毕
        while( UTX0IF == 0 );
        UTX0IF = 0;
    }
}

//串口发送寄存器U0DBUF的数据
void UartTX_Send_Data( unsigned char Data, int len )
{
    for( int j = 0; j < len; j++ )
    {
        U0DBUF = Data;
        while( UTX0IF == 0 );
        UTX0IF = 0;
    }
}

//UART 0中断函数
#pragma vector=URX0_VECTOR
__interrupt void uart0( void )
{
    URX0IF = 0;                 //清中断标志
    P0_0 = ~P0_0;
    uart_buffer = U0DBUF;
    //UartTX_Send_String("welcome",10);
    UartTX_Send_Data( uart_buffer, 1 );
}

//[cnplaman：封装]
void initUART( void )
{
    PERCFG = 0x00;              //外设控制寄存器UART0：P0口位置1
    P0SEL = 0x3c;               //0011 1100：P0_2，P0_3 用作串口(外设功能)

    U0CSR |= 0x80;              //UART方式

    U0GCR |= 10;                //baud_e = 10;
    U0BAUD |= 216;              //波特率设为57600

    UTX0IF = 1;                 //UART0 TX中断标志初始置1[cnplaman]
    U0CSR |= 0x40;              //允许接收
    IEN0 |= 0x84;               //开总中断，允许接收中断
}

void initIO( void )
{
    P0DIR |= 0x01;  //设置P0.0为输出方式[P0.4和P0.5为输入方式???]
    P2DIR |= 0x01;  //设置P2.0为输出方式
    P0_0 = 1;
    P2_0 = 1;
}

void main( void )
{
    unsigned char buf[4];

    initIO();

    CLKCONCMD &= ~0x40;                 //选择32M晶振
    while( !( SLEEPSTA & 0x40 ) );      //等待XSOC稳定
    CLKCONCMD = 0xb8;                   //TICHSPD 128分频，CLKSPD 不分频
    SLEEPCMD |= 0x04;                   //关闭不用的RC 振荡器[cnplaman:not MUST]

    initUART();

    buf[0] = '\t';
    while( 1 )
    {
        P2_0 = ~P2_0;                           //P2_0状态翻转
        delay( 1500 );
        if( uart_buffer == 0x01 )               //按照16进制输入[cnplaman]
        {
            UartTX_Send_String( "welc", 4 ); //串口接收要发送的数据
            UartTX_Send_String( &buf[0], 1 );   //串口发送接收的数据
        }
    }
}

实验步骤
1.创建IAR工作空间；
2.创建基于C的IAR工程；
3.编辑代码并调试；可参考提供的源码，调试应无差错、无告警；
4.试验箱上电；使用mini USB分别连接CC Debugger和PC；并将仿真器CC Debugger的另外一端和ZigBee通用节点板上的JTAG插座相连；
5.按下仿真器CC Debugger的复位按钮RESET，绿灯常亮表示模块连接成功，可以往芯片中烧写程序；每次烧写程序都需要复位，确保仿真器是绿灯状态；
6.在IAR中点击下载并调试按钮download and debug(CTRL+D)烧写程序；
7.在IAR的调试窗口中，点击[全速]运行；
8.用mini USB线连接节点板mini USB和PC机，在PC设备管理器中确认串口端口；
9.打开PC机上的串口工具，选择对应的串口，波特率为57600bps，8位数据、1位停止、无校验、无流控；
10.在超级终端上每发送一次十六进制0x1，CC 2530就会将”Welcome”通过串口传到超级终端；
开发过程中，请注意CTRL+S保存项目！
