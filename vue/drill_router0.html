单页路由
routes: [
  {
    path: '/',
    name: 'home',
    component: Home,
    meta: {title: "home"}
  },
  {
    path: '/home',
    redirect: '/',
    meta: {title: "home"}
  },
  {
    path: '/team',
    name: 'team',
    component: () => import('../components/sub_view/Team.vue'),
    meta: {title: "team"}
  },
  {
    path: '/work',
    name: 'work',
    component: () => import('../components/sub_view/Work.vue'),
    meta: {title: "work"}
  },
  {
    path: '/about',
    name: 'about',
    component: () => import('../components/sub_view/About.vue'),
    meta: {title: "about"}
  },
  {
    path: '/:pathMatch(.*)*',
    name: 'NotFound',
    component: () => import('../components/NotFound.vue'),
    meta: {title: "404"}
  }
]
